WITH interseccion AS (
  SELECT ST_Buffer(ST_Intersection(v1.geom, v2.geom), 15) AS geom
  FROM ft_vias v1
  JOIN ft_vias v2
    ON ST_Intersects(v1.geom, v2.geom)
  WHERE v1.nom_calle = 'MAGALLANES'
	    AND v2.nom_calle = 'COLONIA'
)
SELECT m.*
FROM ft_manzanas m
JOIN interseccion i
  ON ST_Intersects(m.geom, i.geom);
